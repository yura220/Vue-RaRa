라우팅 경로
/                                ← 메인
├── /event                       ← 이벤트 전용 페이지
├── /[weatherType]               ← 날씨별 콘텐츠 선택 화면 (코디/활동 목록)
│   └── /cody                    ← 코디 상세 보기
│          └── /item/:itemId     ← 아이템 상세 보기

├── App.vue          ← 페이지 단위 컴포넌트
├── components/     ← 재사용 가능한 UI 컴포넌트들
│   ├── Home.vue
│   ├── Weather.vue
│   ├── CodyList.vue
│   ├── CodyDetail.vue
│   ├── ItemList.vue
│   ├── Event.vue
│   └── css/
│        ├── main.css
│        ├── weather.css
│        ├── codyList.css
│        ├── codyDetail.css
│        ├── itemList.css
│        ├── event.css
│        └── common.css
├── public/            ← 이미지 저장
│   └── img/
│        └── svg/
├── router/
│   └── index.js
